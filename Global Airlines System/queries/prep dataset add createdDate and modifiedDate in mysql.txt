CREATE DATABASE global_airlines_system;

USE global_airlines_system;

/* adding columns- createdDate and modifiedDate in all tables and adding the today's date to it */

/* airlines table */
ALTER TABLE airlines 
ADD createdDate DATETIME DEFAULT CURRENT_TIMESTAMP,
ADD modifiedDate DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP;

UPDATE airlines
SET createdDate = CURRENT_TIMESTAMP,
    modifiedDate = CURRENT_TIMESTAMP;

SELECT * FROM airlines;
